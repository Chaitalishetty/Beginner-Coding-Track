#Check whether the number is armstrong or not.
Input: N
Output: True or False

#Example:
Input: 153
Output: True

Input: 371
Output: True

Input: 167
Oupput: False

CODE :

#include<iostream>
using namespace std;
 
int main()  
{  
int num,x;
int sum = 0;    
 
cout << "Enter a number:";    
cin >> num;
int i = num;
 
while(i!=0)  
{  
x = i%10;  
sum= sum + x*x*x;  
i = i/10;  
}
 
if(num==sum)  
cout << num << " is an Armstrong Number.";  
else  
cout << num << " is not an Armstrong Number.";  
return 0;
}
